// --- СИСТЕМА СОХРАНЕНИЯ --- // Функция для записи данных в память браузера function saveGame() { localStorage.setItem('clicker_v12_save', JSON.stringify(state)); console.log("Игра сохранена!"); } // Функция для загрузки данных при старте function loadGame() { const savedData = localStorage.getItem('clicker_v12_save'); if (savedData) { const loadedState = JSON.parse(savedData); // Смешиваем старое состояние с загруженным, чтобы не сломать игру при обновлениях state = { ...state, ...loadedState }; // Визуально помечаем купленные узлы в дереве state.boughtNodes.forEach(id => { const node = document.getElementById('node' + id); if (node) node.classList.add('bought'); }); // Проверяем секретные магазины if (state.boughtNodes.includes(1)) document.getElementById('secretShop').style.display = 'block'; if (state.boughtNodes.includes(2)) document.getElementById('x5Shop').style.display = 'block'; updateUI(); } } // Авто-сохранение каждые 30 секунд setInterval(saveGame, 30000); // Запускаем загрузку при открытии страницы window.onload = loadGame; // Добавим в функцию updateUI вызов сохранения, чтобы прогресс не терялся при покупках const originalUpdateUI = updateUI; updateUI = function() { originalUpdateUI(); saveGame(); };
